static setValue(Type type, String name, dynamic value) {
  Map<String, dynamic> properties = _staticProperties[type];
  if (properties == null) {
    properties = new Map<String, dynamic>();
    _staticProperties[type] = properties;
  }

  properties[name] = value;
}
